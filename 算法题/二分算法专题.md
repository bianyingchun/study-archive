## 标准

```js
int binarySearch(int[] nums, int target) {
    int left = 0;
    int right = nums.length - 1; // 注意

    while(left <= right) { // 一定是<=
        int mid = left + (right - left) / 2;
        if(nums[mid] == target)
            return mid;
        else if (nums[mid] < target)
            left = mid + 1; // 注意
        else if (nums[mid] > target)
            right = mid - 1; // 注意
    }
    return -1;
}

```

## 左边界

```js
function findFirstPosition(nums, target) {
  let left = 0;
  let right = nums.length - 1;
  while (left < right) {
    // 取左中位数
    let mid = left + Math.floor((right - left) / 2);
    // 小于一定不是解
    if (nums[mid] < target) {
      // 下一轮搜索区间是 [mid + 1, right]
      left = mid + 1;
    } else if (nums[mid] == target) {
      // 下一轮搜索区间是 [left, mid]
      right = mid;
    } else {
      // nums[mid] > target，下一轮搜索区间是 [left, mid - 1]
      right = mid - 1;
    }
  }

  if (nums[left] == target) {
    return left;
  }
  return -1;
}
```

## 右边界

```js
function findLastPosition(nums, target) {
  let left = 0;
  let right = nums.length - 1;
  while (left < right) {
    // 取右中位数
    let mid = left + Math.ceil((right - left) / 2);
    if (nums[mid] > target) {
      // 下一轮搜索区间是 [left, mid - 1]
      right = mid - 1;
    } else if (nums[mid] == target) {
      // 下一轮搜索区间是 [mid, right]
      left = mid;
    } else {
      // nums[mid] < target，下一轮搜索区间是 [mid + 1, right]
      left = mid + 1;
    }
  }
  return left;
}
```
